title:欢迎来到晚闻子的网页

下面是一款基于python中pygame的猜题游戏

#coding=UTF-8
#导入游戏库
import pygame,sys
from pygame.locals import *
from trash import Sky
class Trivia(object):
#游戏板块初始化
    def __init__(self,filename):
        self.data=[]
        self.current=0
        self.total=0
        self.correct=0
        self.score=0
        self.scored=False
        self.failed=False
        self.wronganswer=0
        self.colors=[white,white,white,white]
#读取问题表的.txt文件
        #read trivia data from file
        f=open(filename,'r',encoding='utf8')
        trivia_data=f.readlines()
        trivia_data=trivia_data
        f.close()
#逐行读取问题，并显示
        #count and clean uo trivia data
        for text_line in trivia_data:
            self.data.append(text_line.strip())
            self.total+=1
#设置游戏背景颜色
    def print_text(font,x,y,text,color=(255,255,255),shadom=True):
        if shadom:
            imgText=font.render(text,True,(0,0,0))
            screen.blit(imgText,(x-2,y-2))
        imgText=font.render(text,True,color)
        screen.blit(imgText,(x,y))
#对游戏问题进行设置
    def show_question(self):
        print_text(font1,260,5,"TRIVIA GAME")
        print_text(font2,240,420,"Press Keys (1-4) To Answer ")
        print_text(font2,700,5,"SCORE ",purple)
        print_text(font2,750,25,str(self.score),purple)
        print_text(font2,623,50,"p=pause l=unpause",purple)
        #get correct answer out of data (first)
        self.correct=int(self.data[self.current+5])
#绘制出游戏文字
        #display question
        question=self.current //6+1
        print_text(font1,55,80,"QUESTION"+str(question))
        print_text(font2,70,120,self.data[self.current])
#当选择正确时则显示绿色，错误显示红色
        #respond to correct answer
        if self.scored:
            self.colors=[white,white,white,white]
            self.colors[self.correct-1]=green
            print_text(font1,280,320,"CORRECT!",green)
            print_text(font2,250,380,"Press Enter For Next Question",green)
        elif self.failed:
            self.colors=[white,white,white,white]
            self.colors[self.wronganswer-1]=red
            self.colors[self.correct-1]=green
            print_text(font1,270,320,"INCORRECT!",red)
            print_text(font2,250,380,"Press Enter For Next Question",red)
        elif self.score==4:
            print_text(font2,300,300,"You Are Great,Fighting!",red)
#显示游戏答案
        #display answers
        print_text(font1,55,170,"ANSWERS")
        print_text(font2,70,210,"1-"+self.data[self.current+1],self.colors[0])
        print_text(font2, 70, 240, "2-" + self.data[self.current + 2], self.colors[1])
        print_text(font2, 70, 270, "3-" + self.data[self.current + 3], self.colors[2])
        print_text(font2, 70, 300, "4-" + self.data[self.current + 4], self.colors[3])
    def handle_input(self,number):
        if not self.scored and not self.failed:
            if number==self.correct:
                self.scored=True
                self.score+=1
            else:
                self.failed=True
                self.wronganswer=number
        if self.score==5:
            self.score=0
#按下空格键则显示下一题
    def next_question(self):
        if self.scored or self.failed:
            self.scored=False
            self.failed=False
            self.correct=0
            self.colors=[white,white,white,white]
            self.current+=6
            if self.current>=self.total:
                self.current=0
#main program begins
pygame.init()
pygame.mixer.init()
screen=pygame.display.set_mode((800,450))
pygame.display.set_caption("晚闻子的真心话冒险")
sky=Sky(screen)
font1=pygame.font.SysFont('arial',40)
font2=pygame.font.SysFont('arial',24)
white=255,255,255
cyan=0,255,255
yellow=255,0,255
purple=255,0,255
green=0,255,0
red=255,0,0
trivia = Trivia(r"C:\Users\Lenovo\OneDrive\桌面\爬虫\喜欢.txt")
pygame.mixer.music.load(r"D:\CloudMusic\Days.mp3")
def print_text(font, x, y, text, color=(255, 255, 255), shadom=True):
    if shadom:
        imgText = font.render(text, True, (0, 0, 0))
        screen.blit(imgText, (x - 2, y - 2))
    imgText = font.render(text, True, color)
    screen.blit(imgText, (x, y))
#repeating loop
while True:
    if pygame.mixer.music.get_busy()==False:
        pygame.mixer.music.play()
    for event in pygame.event.get():
        if event.type==QUIT:
            sys.exit()
        elif event.type==KEYUP:
            if event.key==pygame.K_ESCAPE:
                sys.exit()
            elif event.key==pygame.K_1:
                trivia.handle_input(1)
            elif event.key==pygame.K_2:
                trivia.handle_input(2)
            elif event.key==pygame.K_3:
                trivia.handle_input(3)
            elif event.key==pygame.K_4:
                trivia.handle_input(4)
            elif event.key==pygame.K_RETURN:
                trivia.next_question()
        elif event.type==KEYDOWN:
            if event.key==pygame.K_p:
                pygame.mixer.music.pause()
            elif event.key==pygame.K_l:
                pygame.mixer.music.unpause()
        #clean the screen
        screen.fill((0,255,0))
        sky.blitme()
        #display trivia data
        trivia.show_question()
        #update the display
        pygame.display.update()

一下方法让你的浏览器使用效率极大提高

方法一：完全匹配关键字法“”
即可完全匹配到关键字。其所搜索到的结果，是展示这个完整关键字的结果。而非完整关键字中的某个词。在浏览器搜索栏输入双引号和你所要搜索的内容。如下所示

“ 天气之子 “

方法二。限定标题包含关键字intitle:
方法:intitle:查询词(intitle 加上 英文冒号 加上 搜索关键词)。比如搜索冯羿.video
所发布video视频。因为我们平时搜索相关资讯时，常常会冒出一些乱七八糟的内容，这个方法有效过滤多余内容。

intitle:冯羿.video
